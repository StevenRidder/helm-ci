# ---- Chart spike: headless S-57 ENC -> PNG (DC path, no GL) ----------------
# Append to /tmp/opencpn/cli/CMakeLists.txt (after the helm-spike block).
#
# Compiles a hand-picked slice of gui/src (the s57chart render subgraph) plus
# chart_spike.cpp + chart_stubs.cpp, links the in-tree static libs + model-src
# + wxWidgets. ocpnUSE_GL is left UNDEFINED so the GL branches drop out.

set(CHART_SPIKE_GUI_SRC
  ${PROJECT_SOURCE_DIR}/gui/src/s57chart.cpp
  ${PROJECT_SOURCE_DIR}/gui/src/s57obj.cpp
  # NOTE: gui/src/s57_ocpn_utils.cpp is intentionally NOT compiled. It is a dead
  # file (only its header is in the real GUI_SRC) and it RE-defines the three
  # s57_*LightSectors() helpers that s57chart.cpp already defines -> duplicate
  # symbols, plus it has a pi_rule_list type mismatch. The render path needs
  # none of its symbols.
  ${PROJECT_SOURCE_DIR}/gui/src/senc_manager.cpp
  ${PROJECT_SOURCE_DIR}/gui/src/o_senc.cpp
  ${PROJECT_SOURCE_DIR}/gui/src/ocpndc.cpp
  ${PROJECT_SOURCE_DIR}/gui/src/viewport.cpp
  ${PROJECT_SOURCE_DIR}/gui/src/ocpn_region.cpp
  ${PROJECT_SOURCE_DIR}/gui/src/color_handler.cpp
  # ocpnUSE_GL is defined for the whole tree (and is BAKED into the prebuilt
  # libs' ABI -- e.g. s52plib has GL-only members), so we keep it ON and instead
  # satisfy the GL globals these files reference. These three are small, self-
  # contained TUs that DEFINE those globals + the DC pixel cache:
  ${PROJECT_SOURCE_DIR}/gui/src/ocpn_pixel.cpp       # ocpnMemDC, ocpnBitmap (DC path)
  ${PROJECT_SOURCE_DIR}/gui/src/shaders.cpp          # shader-program globals, preamble
  ${PROJECT_SOURCE_DIR}/gui/src/ocpn_gl_options.cpp  # g_GLOptions
)

add_executable(chart-spike
  chart_spike.cpp
  chart_stubs.cpp
  ${CHART_SPIKE_GUI_SRC}
)
set_property(TARGET chart-spike PROPERTY ENABLE_EXPORTS ON)

target_include_directories(chart-spike PRIVATE
  ${PROJECT_SOURCE_DIR}/include
  ${PROJECT_SOURCE_DIR}/gui/include
  ${PROJECT_SOURCE_DIR}/gui/include/gui          # headers are reached unprefixed
  ${PROJECT_SOURCE_DIR}/gui/src/s57/include
  ${PROJECT_SOURCE_DIR}/libs/s52plib/src
  ${PROJECT_SOURCE_DIR}/libs/s57-charts/include
  ${PROJECT_SOURCE_DIR}/libs/geoprim/include
  ${PROJECT_SOURCE_DIR}/libs/ssl_sha1/include    # s57chart.cpp -> "ssl/sha1.h"
)

# USE_MOCK_DEFS mirrors opencpn-cmd; ocpnUSE_GL deliberately NOT defined.
target_compile_definitions(chart-spike PUBLIC USE_MOCK_DEFS)

# In-tree static libs (each carries its own INTERFACE deps: gdal/tess2/pugixml).
target_link_libraries(chart-spike PRIVATE
  ocpn::s52plib
  ocpn::s57-charts
  ocpn::geoprim
  ocpn::iso8211
  ocpn::gl-headers
)

# Model core (defines g_bopengl, g_bDebugS57, g_bGDAL_Debug, georef, cutil, ...).
target_link_libraries(chart-spike PRIVATE ocpn::model-src ocpn::model)

# sha1 (s57chart::buildSENCName uses ssl/sha1).
target_link_libraries(chart-spike PRIVATE ssl::sha1)

# `typeinfo for ChartPlugInWrapper` is referenced only by a never-executed
# dynamic_cast in the light-sector helper. Emitting it the normal way would drag
# the BSB chart reader (chartimg.cpp). chart_typeinfo.cpp emits ONLY that
# type_info symbol via a minimal stand-in class of the same name (it does NOT
# include the real ChartPlugInWrapper headers).
target_sources(chart-spike PRIVATE chart_typeinfo.cpp)

# wxWidgets last.
target_link_libraries(chart-spike PRIVATE ${wxWidgets_LIBRARIES})
if (TARGET ocpn::wxcurl)
  target_link_libraries(chart-spike PRIVATE ocpn::wxcurl)
else ()
  target_link_libraries(chart-spike PRIVATE ${SYS_WXCURL_LIBRARIES})
endif ()
if (TARGET ocpn::wxsvg)
  target_link_libraries(chart-spike PRIVATE ocpn::wxsvg)
endif ()
target_link_libraries(chart-spike PRIVATE
  ocpn::serial ocpn::tinyxml ocpn::filesystem)
if (HAVE_LIBUDEV)
  target_link_libraries(chart-spike PRIVATE ocpn::libudev)
endif ()
if (MSVC)
  target_link_libraries(chart-spike PRIVATE setupapi.lib psapi.lib iphlpapi)
endif ()
