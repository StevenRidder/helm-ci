# ---- ocpn::chart-render — first-class static library for the headless S-52
#      renderer (Step 6). Was a per-executable gui/src slice + stubs; now a
#      reusable library that chart-spike AND helm-tiles link. OCPN_HEADLESS
#      severs the plugin dynamic_cast, so NO faked RTTI (chart_typeinfo.cpp) is
#      needed. PUBLIC usage requirements (includes/defs/libs) flow to consumers,
#      and GetpSharedDataLocation() lives in chart_stubs.cpp so the library has
#      no back-reference to its host executable.
add_library(helm-chartrender STATIC ${CHART_SPIKE_GUI_SRC} chart_stubs.cpp)
target_include_directories(helm-chartrender PUBLIC
  ${PROJECT_SOURCE_DIR}/include
  ${PROJECT_SOURCE_DIR}/gui/include
  ${PROJECT_SOURCE_DIR}/gui/include/gui
  ${PROJECT_SOURCE_DIR}/gui/src/s57/include
  ${PROJECT_SOURCE_DIR}/libs/s52plib/src
  ${PROJECT_SOURCE_DIR}/libs/s57-charts/include
  ${PROJECT_SOURCE_DIR}/libs/geoprim/include
  ${PROJECT_SOURCE_DIR}/libs/ssl_sha1/include)
target_compile_definitions(helm-chartrender PUBLIC USE_MOCK_DEFS OCPN_HEADLESS)
target_link_libraries(helm-chartrender PUBLIC
  ocpn::s52plib ocpn::s57-charts ocpn::geoprim ocpn::iso8211 ocpn::gl-headers
  ocpn::model-src ocpn::model ssl::sha1
  ocpn::serial ocpn::tinyxml ocpn::filesystem
  ${wxWidgets_LIBRARIES})
if (TARGET ocpn::wxcurl)
  target_link_libraries(helm-chartrender PUBLIC ocpn::wxcurl)
endif ()
if (TARGET ocpn::wxsvg)
  target_link_libraries(helm-chartrender PUBLIC ocpn::wxsvg)
endif ()
if (HAVE_LIBUDEV)
  target_link_libraries(helm-chartrender PUBLIC ocpn::libudev)
endif ()
add_library(ocpn::chart-render ALIAS helm-chartrender)

# chart-spike: the proof harness, now just a main linking the library.
add_executable(chart-spike chart_spike.cpp)
set_property(TARGET chart-spike PROPERTY ENABLE_EXPORTS ON)
target_link_libraries(chart-spike PRIVATE ocpn::chart-render)
