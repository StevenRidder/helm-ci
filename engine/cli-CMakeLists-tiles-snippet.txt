# ---- Helm tile server: S-52 ENC -> HTTP tiles (clones the chart-spike slice) ----
add_executable(helm-tiles
  helm_tiles.cpp
  chart_stubs.cpp
  chart_typeinfo.cpp
  ${CHART_SPIKE_GUI_SRC}
)
set_property(TARGET helm-tiles PROPERTY ENABLE_EXPORTS ON)
target_include_directories(helm-tiles PRIVATE
  ${PROJECT_SOURCE_DIR}/include
  ${PROJECT_SOURCE_DIR}/gui/include
  ${PROJECT_SOURCE_DIR}/gui/include/gui
  ${PROJECT_SOURCE_DIR}/gui/src/s57/include
  ${PROJECT_SOURCE_DIR}/libs/s52plib/src
  ${PROJECT_SOURCE_DIR}/libs/s57-charts/include
  ${PROJECT_SOURCE_DIR}/libs/geoprim/include
  ${PROJECT_SOURCE_DIR}/libs/ssl_sha1/include
  ${PROJECT_SOURCE_DIR}/libs/IXWebSocket
)
target_compile_definitions(helm-tiles PUBLIC USE_MOCK_DEFS)
target_link_libraries(helm-tiles PRIVATE
  ocpn::s52plib ocpn::s57-charts ocpn::geoprim ocpn::iso8211 ocpn::gl-headers)
target_link_libraries(helm-tiles PRIVATE ocpn::model-src ocpn::model)
target_link_libraries(helm-tiles PRIVATE ssl::sha1)
target_link_libraries(helm-tiles PRIVATE ixwebsocket)
target_link_libraries(helm-tiles PRIVATE ${wxWidgets_LIBRARIES})
if (TARGET ocpn::wxcurl)
  target_link_libraries(helm-tiles PRIVATE ocpn::wxcurl)
else ()
  target_link_libraries(helm-tiles PRIVATE ${SYS_WXCURL_LIBRARIES})
endif ()
if (TARGET ocpn::wxsvg)
  target_link_libraries(helm-tiles PRIVATE ocpn::wxsvg)
endif ()
target_link_libraries(helm-tiles PRIVATE ocpn::serial ocpn::tinyxml ocpn::filesystem)
if (HAVE_LIBUDEV)
  target_link_libraries(helm-tiles PRIVATE ocpn::libudev)
endif ()
